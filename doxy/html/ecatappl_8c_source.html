345void ECAT_CheckTimer(void)
346{
347

349 if(bEcatWaitForAlControlRes && (EsmTimeoutCounter > 0))
350 {
351 EsmTimeoutCounter--;
352 }

355
356 DC_CheckWatchdog();

359/*ECATCHANGE_START(V5.12) COE4*/
360
361 /* Increment the counter every ms between two updates based on the system time (32Bit overrun is handled in COE_SyncTimeStamp) */
362 if (!b32BitDc || ((u64Timestamp & 0xFFFFFFFF) <= 4293000000))
363 {
364
365 /* the timestamp is stored in ns */
366 u64Timestamp = u64Timestamp + 1000000;
367

372 u32CheckForDcOverrunCnt = CHECK_DC_OVERRUN_IN_MS;

380/////////////////////////////////////////////////////////////////////////////////////////
381/**
382\brief In case of non DC synchronization the cycle time measurement is started and 0x1C3.2 (Cycle time) is updated
383*////////////////////////////////////////////////////////////////////////////////////////
384void HandleCycleTimeMeasurement(void)
385{
386 if (!bDcSyncActive) //no DC sync configured (cycle time measurement 0x1C3x.2 is only available in no DC sync modes)
387 {
388 if (u32CycleTimeStartValue > 0)
389 {
390 /* bus cycle completed*/
391 u32CycleTimeStartValue = GetSystemTimeDelay(u32CycleTimeStartValue);
392
393 if ((sSyncManOutPar.u32CycleTime == 0) || (sSyncManOutPar.u32CycleTime > u32CycleTimeStartValue))
394 {
395 sSyncManOutPar.u32CycleTime = u32CycleTimeStartValue;
396 }
397
398 if ((sSyncManInPar.u32CycleTime == 0) || (sSyncManInPar.u32CycleTime > u32CycleTimeStartValue))
399 {
400 sSyncManInPar.u32CycleTime = u32CycleTimeStartValue;
401 }
402 }
403 /* get next start value */
404 u32CycleTimeStartValue = GetSystemTimeDelay(0);
405
406 }/* No DC sync configured */
407}

409/////////////////////////////////////////////////////////////////////////////////////////
410/**
411\param u32StartTime Old system time (0x910:0x913) value
412
413\return System time delta in ns
414
415\brief Calculates the difference between the old and current system time value in ns.
416 NOTE: This function only handles a 32Bit system time values (therefore the maximum delay about 4sec).
417*////////////////////////////////////////////////////////////////////////////////////////
418
419UINT32 GetSystemTimeDelay(UINT32 u32StartTime)
420{
421 UINT32 u32CurValue = 0;
422 UINT32 u32Delta = 0;
423
424
425 HW_EscReadDWordIsr(u32CurValue, ESC_SYSTEMTIME_OFFSET);
426
427 if (u32CurValue > 0)
428 {
429 if (u32StartTime <= u32CurValue)
430 {
431 u32Delta = u32CurValue - u32StartTime;
432 }
433 else
434 {
435 //The 32Bit timer is wrapped around
436 u32Delta = u32CurValue + (0xFFFFFFFF - u32StartTime);
437 }
438 }// current value successfully read out
439
440 if (u32StartTime > 0)
441 {
442
443 /*the difference between two timestamps are calculated => subtract measurement failure*/
444 if (u32SystemTimeReadFailure < u32Delta)
445 {
446 u32Delta = u32Delta - u32SystemTimeReadFailure;
447 }
448 else
449 {
450 /*set the delta to 0 if the measurement failure is greater than the calculated difference*/
451 u32Delta = 0;
452 }
453 }
454 return u32Delta;
455}

458void PDI_Isr(void)
459{
460 if(bEscIntEnabled)
461 {
462 /* get the AL event register */
463 UINT16 ALEvent = HW_GetALEventRegister_Isr();
464 ALEvent = SWAPWORD(ALEvent);
465
466 if ( ALEvent & PROCESS_OUTPUT_EVENT )
467 {
468 if(bDcRunning && bDcSyncActive)
469 {
470 /* Reset SM/Sync0 counter. Will be incremented on every Sync0 event*/
471 u16SmSync0Counter = 0;
472 }
473 if(sSyncManOutPar.u16SmEventMissedCounter > 0)
474 {
475 sSyncManOutPar.u16SmEventMissedCounter--;
476 }
477
478/*ECATCHANGE_START(V5.12) ECAT5*/
479 sSyncManInPar.u16SmEventMissedCounter = sSyncManOutPar.u16SmEventMissedCounter;
480/*ECATCHANGE_END(V5.12) ECAT5*/
481
482
483
484 /* Outputs were updated, set flag for watchdog monitoring */
485 bEcatFirstOutputsReceived = TRUE;
486
487
488 /*
489 handle output process data event
490 */
491 if ( bEcatOutputUpdateRunning )
492 {
493 /* slave is in OP, update the outputs */
494 PDO_OutputMapping();
495 }
496 else
497 {
498 /* Just acknowledge the process data event in the INIT,PreOP and SafeOP state */
499 HW_EscReadByteIsr(u8dummy,nEscAddrOutputData);
500 HW_EscReadByteIsr(u8dummy,(nEscAddrOutputData+nPdOutputSize-1));
501 }
502 }
503
504 /*
505 Call ECAT_Application() in SM Sync mode
506 */
507 if (sSyncManOutPar.u16SyncType == SYNCTYPE_SM_SYNCHRON)
508 {
509 /* The Application is synchronized to process data Sync Manager event*/
510 ECAT_Application();
511 }
512
513 if ( bEcatInputUpdateRunning
514 && ((sSyncManInPar.u16SyncType == SYNCTYPE_SM_SYNCHRON) || (sSyncManInPar.u16SyncType == SYNCTYPE_SM2_SYNCHRON))
515 )
516 {
517 /* EtherCAT slave is at least in SAFE-OPERATIONAL, update inputs */
518 PDO_InputMapping();
519 }
520
521 /*
522 Check if cycle exceed
523 */
524 /*if next SM event was triggered during runtime increment cycle exceed counter*/
525 ALEvent = HW_GetALEventRegister_Isr();
526 ALEvent = SWAPWORD(ALEvent);
527
528 if ( ALEvent & PROCESS_OUTPUT_EVENT )
529 {
530 sSyncManOutPar.u16CycleExceededCounter++;
531 sSyncManInPar.u16CycleExceededCounter = sSyncManOutPar.u16CycleExceededCounter;
532
533 /* Acknowledge the process data event*/
534 HW_EscReadByteIsr(u8dummy,nEscAddrOutputData);
535 HW_EscReadByteIsr(u8dummy,(nEscAddrOutputData+nPdOutputSize-1));
536 }
537 } //if(bEscIntEnabled)
538
539/*ECATCHANGE_START(V5.12) ECAT5*/
540 COE_UpdateSyncErrorStatus();
541/*ECATCHANGE_END(V5.12) ECAT5*/
542
543}

545void Sync0_Isr(void)
546{
547 Sync0WdCounter = 0;
548
549 if(bDcSyncActive)
550 {
551
552 if ( bEcatInputUpdateRunning )
553 {
554 LatchInputSync0Counter++;
555 }
556
557 if(u16SmSync0Value > 0)
558 {
559 /* Check if Sm-Sync sequence is invalid */
560 if (u16SmSync0Counter > u16SmSync0Value)
561 {
562 if ((nPdOutputSize > 0) && (sSyncManOutPar.u16SmEventMissedCounter <= sErrorSettings.u16SyncErrorCounterLimit))
563 {
564 sSyncManOutPar.u16SmEventMissedCounter = sSyncManOutPar.u16SmEventMissedCounter + 3;
565 }
566
567 if ((nPdInputSize > 0) && (nPdOutputSize == 0) && (sSyncManInPar.u16SmEventMissedCounter <= sErrorSettings.u16SyncErrorCounterLimit))
568 {
569 sSyncManInPar.u16SmEventMissedCounter = sSyncManInPar.u16SmEventMissedCounter + 3;
570 }
571
572 } // if (u16SmSync0Counter > u16SmSync0Value)
573
574
575 if ((nPdOutputSize == 0) && (nPdInputSize > 0))
576 {
577 /* Input only with DC, check if the last input data was read*/
578 UINT16 ALEvent = HW_GetALEventRegister_Isr();
579 ALEvent = SWAPWORD(ALEvent);
580
581 if ((ALEvent & PROCESS_INPUT_EVENT) == 0)
582 {
583
584 /* no input data was read by the master, increment the sm missed counter*/
585 if (u16SmSync0Counter <= u16SmSync0Value)
586 {
587 u16SmSync0Counter++;
588 }
589 }
590 else
591 {
592 /* Reset SM/Sync0 counter*/
593 u16SmSync0Counter = 0;
594
595 sSyncManInPar.u16SmEventMissedCounter = 0;
596
597 }
598 }
599 else if (u16SmSync0Counter <= u16SmSync0Value)
600 {
601
602 u16SmSync0Counter++;
603 }
604 }//SM -Sync monitoring enabled
605
606
607 /* Application is synchronized to SYNC0 event*/
608 ECAT_Application();
609
610 if ( bEcatInputUpdateRunning
611 && (LatchInputSync0Value > 0) && (LatchInputSync0Value == LatchInputSync0Counter) ) /* Inputs shall be latched on a specific Sync0 event */
612 {
613 /* EtherCAT slave is at least in SAFE-OPERATIONAL, update inputs */
614 PDO_InputMapping();
615
616 if(LatchInputSync0Value == 1)

945/////////////////////////////////////////////////////////////////////////////////////////
946/**
947 \param wordaddress WORD based address in the EEPROM to be read
948 \param wordsize size in WORD of EEPROM data to be read
949 \param pData buffer for EEPROM to be read or with EEPROM data to be written
950 \param access read (ESC_RD) or write (ESC_WR) access
951
952 \return 0 - success, > 0: error code
953
954 \brief This function is called to read or write the EEPROM data, for BigEndian
955 \brief Controller (switch BIG_ENDIAN_FORMAT set) the data has to be swapped outside
956 \brief of this function)
957*////////////////////////////////////////////////////////////////////////////////////////
958
959UINT16 ESC_EepromAccess(UINT32 wordaddress, UINT16 wordsize, UINT16 MBXMEM *pData, UINT8 access)
960{
961 UINT16 i;
962 UINT8 u8TimeOut;
963 UINT16 u16RetErr = 0;
964 UINT16 u16WordOffset = 0;
965 UINT8 RetryCnt = MAX_CMD_RETIRES; //Maximum number of retries (evaluated in case of an Acknowledge Error)
966
967 UINT16 nConfig;
968 VARVOLATILE UINT16 nControl;
969
970 HW_EscReadWord(nConfig,ESC_EEPROM_CONFIG_OFFSET);
971 HW_EscReadWord(nControl, ESC_EEPROM_CONTROL_OFFSET);
972
973
974
975 if ( nConfig & ESC_EEPROM_ASSIGN_TO_PDI_MASK )
976 {
977 /* register 0x500.0 is set (should be written by the master before sending
978 the state transition request to PREOP),we have access to the EEPROM */
979 UINT16 step = 1; /* we write always only 1 word with one write access */
980
981 if ( access == ESC_RD )
982 {
983 /* read access requested, we have to check if we read 2 (register 0x502.6=0)
984 or 4 words (register 0x502.6=1) with one access */
985 if ( nControl & ESC_EEPROM_SUPPORTED_READBYTES_MASK )
986 {
987 step = 4; /* we get 4 words with one read access */
988 }
989 else
990 {
991 step = 2; /* we get 2 words with one read access */
992 }
993 }
994
995 /* first we have to lock the EEPROM access that we will not be interrupted by the master
996 by setting register 0x501.0 */
997 nConfig = ESC_EEPROM_LOCKED_BY_PDI_MASK;
998 HW_EscWriteWord(nConfig,ESC_EEPROM_CONFIG_OFFSET);
999 for (i = 0; i < wordsize;)
1000 {
1001 /* we have to set the start address in register 0x504-0x507 */
1002 HW_EscWrite((MEM_ADDR *) &wordaddress, ESC_EEPROM_ADDRESS_OFFSET, 4);
1003
1004 if ( access == ESC_RD )
1005 {
1006 /* read access, we start the reading by setting 0x502.8
1007 (will be reset automatically when reading is finished) */
1008 nControl = ESC_EEPROM_CMD_READ_MASK;
1009 HW_EscWriteWord(nControl,ESC_EEPROM_CONTROL_OFFSET);
1010 }
1011 else
1012 {
1013 /* write access, we write the data in register 0x508-0x509 */
1014 HW_EscWriteWord(pData[i],ESC_EEPROM_DATA_OFFSET);
1015
1016 /* we start the writing by setting 0x502.9
1017 (will be reset automatically when writing is finished) */
1018
1019 nControl = SWAPWORD(ESC_EEPROM_CMD_WRITE_MASK);
1020
1021 HW_EscWriteWord(nControl, ESC_EEPROM_CONTROL_OFFSET);
1022 }
1023
1024 do
1025 {
1026
1027 /*Wait 100 cycles before reading EEPROM status*/
1028 u8TimeOut = 100;
1029 while (u8TimeOut > 0)
1030 {
1031 u8TimeOut--;
1032 }
1033 HW_EscReadWord(nControl, ESC_EEPROM_CONTROL_OFFSET);
1034
1035 }
1036 while ( nControl & (ESC_EEPROM_BUSY_MASK));
1037
1038 /* we have to check if the access was without errors */
1039 HW_EscReadWord(nControl, ESC_EEPROM_CONTROL_OFFSET);
1040
1041 if ( nControl & ESC_EEPROM_ERROR_MASK )
1042 {
1043 if(!(nControl & ESC_EEPROM_ERROR_CMD_ACK) && (RetryCnt != 0))
1044 {
1045 /* Only abort if non Acknowledge Error occurs
1046 In case of an Acknowledge Error the operation should be repeated*/
1047 u16RetErr = ALSTATUSCODE_EE_ERROR;
1048 break;
1049 }
1050 }
1051 else
1052 {
1053 if ( access == ESC_RD )
1054 {
1055 UINT16 u16BytesToCopy = (step << 1);
1056
1057 /* read access, get the data from register 0x508-0x50B(0x50F)*/
1058 if((u16WordOffset + step) > wordsize)
1059 {
1060 /*less than "step" words are left => copy only last required Bytes*/
1061 u16BytesToCopy = (wordsize - u16WordOffset) << 1;
1062 }
1063
1064 HW_EscRead((MEM_ADDR *) &pData[i], ESC_EEPROM_DATA_OFFSET, u16BytesToCopy);
1065 }
1066 }
1067
1068 if(!(nControl & ESC_EEPROM_ERROR_MASK))
1069 {
1070 /* In case of Acknowledge Error repeat same operation, otherwise increment the address and proceed*/
1071 wordaddress += step;
1072 u16WordOffset +=step;
1073 RetryCnt = MAX_CMD_RETIRES;
1074
1075 i += step;
1076 }
1077 else
1078 {
1079 RetryCnt --;
1080 if(RetryCnt > 0)
1081 {
1082/*ECATCHANGE_START(V5.12) ECAT8*/
1083 /* Set wait (11ms / retries) until repeat EEPROM access */
1084 INT32 i32TimeoutTicks = (INT32)(ECAT_TIMER_INC_P_MS * (11/ MAX_CMD_RETIRES));
1085/*ECATCHANGE_END(V5.12) ECAT8*/
1086 UINT16 u16CurTimer = 0;
1087 UINT16 u16LastTimer = HW_GetTimer();
1088 UINT16 Delta = 0;
1089
1090 /* Start wait loop */
1091 while(i32TimeoutTicks > 0)
1092 {
1093 u16CurTimer = HW_GetTimer();
1094
1095 if(u16LastTimer < u16CurTimer)
1096 {
1097/*ECATCHANGE_START(V5.12) ECAT8*/
1098 Delta = (u16CurTimer - u16LastTimer);
1099/*ECATCHANGE_END(V5.12) ECAT8*/
1100 }
1101 else
1102 {
1103 /* 16bit overrun*/
1104 Delta = (0xFFFF - u16LastTimer) + u16CurTimer;
1105 }
1106 i32TimeoutTicks = i32TimeoutTicks - Delta;
1107
1108 u16LastTimer = u16CurTimer;
1109 }
1110 }
1111 else
1112 {
1113 /* Abort EEPROM access if max retires are reached*/
1114 u16RetErr = ALSTATUSCODE_EE_ERROR;
1115 break;
1116 }
1117 }
1118 } //for-loop over all data
1119 } // if EEPROM access is assigned to PDI
1120 else
1121 {
1122 u16RetErr = ALSTATUSCODE_EE_NOACCESS;
1123 }
1124
1125 /* clear EEPROM control register 0x500 */
1126 nConfig = 0;
1127
1128 HW_EscWriteWord(nConfig, ESC_EEPROM_CONFIG_OFFSET);
1129 return u16RetErr;
1130}

1133/**
1134 \return success : ALSTATUSCODE_WAITFORCOLDSTART
1135 failed : EEPROM ERRROR Code
1136
1137 \brief This function recalculates the EEPROM CRC and writes the updated value to EEPROM.
1138 After writing the ESC Config Area a device restart is required!
1139 *////////////////////////////////////////////////////////////////////////////////////////
1140UINT16 ESC_EepromWriteCRC(void)
1141{
1142 UINT16 u16Return = ALSTATUSCODE_UNSPECIFIEDERROR;
1143 UINT16 EscCfgData[8];
1144 UINT16 u16Crc = 0x00FF;
1145 UINT16 i,j;
1146
1147
1148 u16Return = ESC_EepromAccess(0,7,(UINT16 *)EscCfgData,ESC_RD);
1149 if(u16Return == 0)
1150 {
1151 UINT8 *pData = (UINT8 *)EscCfgData;
1152
1153 for(i = 0; i < 14; i++ )
1154 {
1155 u16Crc ^= pData[i];
1156
1157 for(j=0; j<8; j++ )
1158 {
1159 if( u16Crc & 0x80 )
1160 {
1161 u16Crc = (u16Crc<<1) ^ 0x07;
1162 }
1163 else
1164 {
1165 u16Crc <<= 1;
1166 }
1167 }
1168 }
1169
1170 /*only low Byte shall be written*/
1171 u16Crc &= 0x00FF;
1172
1173 /*write new calculated Crc to Esc Config area*/
1174 u16Return = ESC_EepromAccess(7,1,&u16Crc,ESC_WR);
1175 if(u16Return == 0)
1176 {
1177 u16Return = ALSTATUSCODE_WAITFORCOLDSTART;
1178 }
1179 }
1180
1181 return u16Return;
1182}
