98 TCiA402Axis *pAxis = pAxisObj;
99 static s32 PreTragetPos = 0;
100 if (PreTragetPos != pAxis->Objects.objTargetPosition)
101 {
102 float Radiant, Duration;
103 Radiant = pAxis->Objects.objHomeOffset*(M_PI/180);//Degree
104 Duration = ((pAxis->Objects.objTargetPosition - MC_GetCurrentPosition1())*1000)/(*(pAxis->Objects.objTargetVelocity));
105 MC_ProgramPositionCommandMotor1(Radiant, Duration);
106 PreTragetPos = *(pAxis->Objects.pobjTargetPosition);
107 }
