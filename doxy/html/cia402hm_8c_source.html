93/*
94 * ----------------------------------------------------------------------------
95 * Function Name: CIA402HM_Run()
96 * Purpose:
97 * Params:
98 * Returns:
99 * Note:
100 * ----------------------------------------------------------------------------
101 */
102s32 CIA402HM_Run(void *pAxisObj)
103{
104 TCiA402Axis *pAxis = pAxisObj;
105 CIA402HM_CTRL_OBJ *pCtrl = pAxis->pFuncIntf->pCtrlObj;
106 CIA402HM_CONTROLWORD_OBJ *pControlWord = (CIA402HM_CONTROLWORD_OBJ*)&(pAxis->Objects.objControlWord);
107 CIA402HM_STATUSWORD_OBJ *pStatusWord = (CIA402HM_STATUSWORD_OBJ*)&(pAxis->Objects.objStatusWord);
108 static UINT16 curTime=0;
109 static u32 timeOut=0, homingDuration=0;
110
111 switch (pCtrl->State)
112 {
113 default:
114 case CIA402HM_IDLE:
115
116 pStatusWord->b.homing_attained = 1;
117 if (pControlWord->b.homing_op_start == 1)
118 {
119 pStatusWord->b.homing_attained = 0;
120 pCtrl->State = CIA402HM_START;
121 }
122 break;
123
124 case CIA402HM_START:
125 /* Start homing */
126 pStatusWord->b.target_reached = 0;
127 pStatusWord->b.internal_limit_active = 0;
128 pStatusWord->b.homing_attained = 0;
129 pStatusWord->b.homing_error = 0;
130 if (CIA402_CheckEmulationEnable() == CIA402_EMUL_DISABLED)
131 {
132 if (pAxis->Objects.objHomingSpeed.SpeedDuringSearchForZero <= 0)
133 {
134 pAxis->Objects.objHomingSpeed.SpeedDuringSearchForZero = 1;
135 }
136 homingDuration = (360000/(pAxis->Objects.objHomingSpeed.SpeedDuringSearchForZero));
137 pPosCtrlM1.AlignmentStatus = TC_AWAITING_FOR_ALIGNMENT;
138 STM_NextState( pMCI[M1]->pSTM, ALIGN_TEST );
139 }
140
141 pCtrl->State = CIA402HM_RUN;
142 curTime = HW_GetTimer();
143 break;
144 case CIA402HM_TEST:
145
146 break;
147 case CIA402HM_RUN:
148 /* Active homing */
149 if (CIA402_CheckEmulationEnable())
150 {
151 if(HW_CheckTimeout(curTime, 4000*2))
152 {
153 PID_HandleInit(&PID_PosParamsM1);
154 pPosCtrlM1.Theta = 0.0f;
155 ENC_SetMecAngle(pPosCtrlM1.pENC , 0);
156 pPosCtrlM1.pENC->_Super.wMecAngle = 0;
157 /* Homing successfully completed */
158 pStatusWord->b.target_reached = 1;
159
160 pAxis->Objects.objPositionActualValue=0;
161
162 pCtrl->State = CIA402HM_DONE;
163 }
164 }
165 else if(MC_GetAlignmentStatusMotor1() == TC_ALIGNMENT_COMPLETED)//if (POS_CheckAlignmentDone())
166 {
167 /* Run offset */
168 if ((pAxis->Objects.objHomeOffset))
169 {
170 if (pAxis->Objects.objHomingSpeed.SpeedDuringSearchForZero <= 0)
171 {
172 pAxis->Objects.objHomingSpeed.SpeedDuringSearchForZero = 1;
173 }
174 homingDuration = (abs(pAxis->Objects.objHomeOffset)*1000/1000)/(pAxis->Objects.objHomingSpeed.SpeedDuringSearchForZero);
175
176 float currentPositionRad = (float)(SPD_GetMecAngle(STC_GetSpeedSensor(pMCI[M1]->pSTC))) / RADTOS16;
177 volatile static float Radiant, Duration;
178 s32 stmp32 = pAxis->Objects.objHomeOffset;
179 Radiant = stmp32*MOTOR_RATIO + currentPositionRad;//Degree
180 Duration = homingDuration*(0.0009f);
181 MC_ProgramPositionCommandMotor1(Radiant, Duration);
182
183 timeOut = homingDuration+1000;
184 pCtrl->State = CIA402HM_RUN_OFFSET;
185 }
186 else
187 {
188 /* Homing successfully completed */
189 pStatusWord->b.target_reached = 1;
190 pCtrl->State = CIA402HM_DONE;
191 }
192 curTime = HW_GetTimer();
193 }
194 if(HW_CheckTimeout(curTime, 5000*2))
195 {
196 pStatusWord->b.target_reached = 0;
197 pStatusWord->b.homing_error = 1;
198
199 /* Report error */
200 pCtrl->State = CIA402HM_DONE;
201 }
202 else if (pControlWord->b.homing_op_start == 0)
203 {
204 PID_SetIntegralTerm(&PID_PosParamsM1, 0x00000000UL);
205 PID_SetPrevError(&PID_PosParamsM1, 0x00000000UL);
206 ENC_SetMecAngle(&ENCODER_M1 , 0);
207 ENCODER_M1._Super.wMecAngle = 0;
208 pPosCtrlM1.Theta = 0.0f;
209 pStatusWord->b.target_reached = 0;
210 pCtrl->State = CIA402HM_DONE;
211 }
212 break;
213
214 case CIA402HM_RUN_OFFSET:
215 if(HW_CheckTimeout(curTime, timeOut*2))
216 {
217
218 PID_SetIntegralTerm(&PID_PosParamsM1, 0x00000000UL);
219 PID_SetPrevError(&PID_PosParamsM1, 0x00000000UL);
220 pPosCtrlM1.MecAngleOffset = pPosCtrlM1.pENC->_Super.hMecAngle;
221 pPosCtrlM1.pENC->_Super.wMecAngle = 0;
222 pPosCtrlM1.Theta = 0.0f;
223 ENC_SetMecAngle(pPosCtrlM1.pENC , pPosCtrlM1.MecAngleOffset);
224 pStatusWord->b.target_reached = 1;
225 pCtrl->State = CIA402HM_DONE;
226 }
227
228 break;
229
230 case CIA402HM_DONE:
231 timeOut = 1000;
232 curTime = HW_GetTimer();
233 pCtrl->State = CIA402HM_WAIT;
234 break;
235
236 case CIA402HM_WAIT:
237 if(HW_CheckTimeout(curTime, timeOut*2))
238 {
239 pStatusWord->b.homing_attained = 1;
240 pCtrl->State = CIA402HM_STOP;
241 }
242 break;
243
244 case CIA402HM_STOP:
245 /* Wait homing_op_start bit to be down */
246 if (pControlWord->b.homing_op_start == 0)
247 {
248 pCtrl->State = CIA402HM_IDLE;
249 }
250 break;
251 }
252 return 0;
253} /* End of CIA402HM_Run() */
