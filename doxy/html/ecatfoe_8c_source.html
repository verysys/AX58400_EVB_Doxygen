87/////////////////////////////////////////////////////////////////////////////////////////
88/**
89 \param pFoeInd Pointer to the received mailbox data from the master.
90
91 \return result of the operation (0 (success) or mailbox error code (MBXERR_.... defined in
92 mailbox.h))
93
94 \brief This function is called when a FoE (File Access over EtherCAT) service is received from
95 the master.
96*////////////////////////////////////////////////////////////////////////////////////////
97
98UINT8 FOE_ServiceInd(TFOEMBX MBXMEM * pFoeInd)
99{
100 /* initialize the result of the service checking */
101 UINT16 nextState = ECAT_FOE_ERRCODE_ILLEGAL;
102 /* dataSize contains the size of the file data */
103 UINT16 dataSize = SWAPWORD(pFoeInd->MbxHeader.Length) - FOE_HEADER_SIZE;
104
105
106 /* it has to be checked if the mailbox protocol is correct, the sent mailbox data length has to
107 great enough for the service header of the FoE service */
108 if ( SWAPWORD(pFoeInd->MbxHeader.Length) < FOE_HEADER_SIZE )
109 {
110 return MBXERR_SIZETOOSHORT;
111 }
112
113 switch ( SWAPWORD(pFoeInd->FoeHeader.OpCode) )
114 {
115 case ECAT_FOE_OPCODE_RRQ:
116 /* file read is requested */
117 if ( u16FileAccessState == FOE_READY )
118 {
119/*ECATCHANGE_START(V5.12) FOE2*/
120 UINT32 u32Password = SWAPDWORD((((UINT32)pFoeInd->FoeHeader.Cmd.Password[FOE_COMMAND_LOWWORD]) | ((UINT32)pFoeInd->FoeHeader.Cmd.Password[FOE_COMMAND_HIGHWORD]) << 16));
121/*ECATCHANGE_END(V5.12) FOE2*/
122 /* last FoE sequence was finished, call application function */
123/* ECATCHANGE_START(V5.12)*/
124 nextState = FOE_Read((UINT16 MBXMEM *)pFoeInd->Data, dataSize, (UINT16*)pFoeInd->Data, u32Password);
125/* ECATCHANGE_END(V5.12)*/
126
127 /* u32LastFileOffset contains the offset of the file which is sent now */
128 u32LastFileOffset = 0;
129
130 if (nextState <= FOE_MAXDATA)
131 {
132 /* the first data packet shall be sent */
133 u32PacketNo = 1;
134 /* u32FileOffset contains the offset of the file which shall be sent when the next FoE ACK is received */
135 u32FileOffset = nextState;
136 }
137 else if (nextState <= FOE_MAXBUSY)
138 {
139 u32PacketNo = 0;
140 u32FileOffset = 0;
141 u16FileAccessState = FOE_WAIT_FOR_ACK;
142 }
143 }
144 break;
145
146 case ECAT_FOE_OPCODE_WRQ:
147 /* file write is requested */
148 if ( u16FileAccessState == FOE_READY )
149 {
150/*ECATCHANGE_START(V5.12) FOE2*/
151 UINT32 u32Password = SWAPDWORD((((UINT32)pFoeInd->FoeHeader.Cmd.Password[FOE_COMMAND_LOWWORD]) | ((UINT32)pFoeInd->FoeHeader.Cmd.Password[FOE_COMMAND_HIGHWORD]) << 16));
152/*ECATCHANGE_END(V5.12) FOE2*/
153 /* last FoE sequence was finished, call application function */
154/* ECATCHANGE_START(V5.12)*/
155 nextState = FOE_Write((UINT16 MBXMEM *)pFoeInd->Data, dataSize, u32Password);
156/* ECATCHANGE_END(V5.12)*/
157 if ( nextState == 0 )
158 {
159 /* checking was successful, sent a FoE Ack service */
160 nextState = FOE_ACK;
161 }
162 /* initialize the packet number */
163 u32PacketNo = 0;
164 }
165 break;
166
167 case ECAT_FOE_OPCODE_DATA:
168 /* file data is received */
169 if ( u16FileAccessState == FOE_WAIT_FOR_DATA
170 || u16FileAccessState == FOE_WAIT_FOR_LAST_DATA )
171 {
172/*ECATCHANGE_START(V5.12) FOE2*/
173 UINT32 u32CmdPacketNo = SWAPDWORD((((UINT32)pFoeInd->FoeHeader.Cmd.PacketNo[FOE_COMMAND_LOWWORD]) | ((UINT32)pFoeInd->FoeHeader.Cmd.PacketNo[FOE_COMMAND_HIGHWORD]) << 16));
174/*ECATCHANGE_END(V5.12) FOE2*/
175
176 /* we are waiting for file data, service is correct */
177 if ( u32CmdPacketNo == u32PacketNo )
178 {
179 /* the packet number is correct, call application function to store the file data */
180/* ECATCHANGE_START(V5.12)*/
181 nextState = FOE_Data((UINT16 MBXMEM *)pFoeInd->Data, dataSize);
182/* ECATCHANGE_END(V5.12)*/
183 if ( nextState == 0 )
184 {
185 /* checking was successful, sent a FoE Ack service */
186 nextState = FOE_ACK;
187 }
188 }
189 else
190 {
191 nextState = ECAT_FOE_ERRCODE_PACKENO;
192 }
193 }
194 break;
195
196 case ECAT_FOE_OPCODE_ACK:
197 /* acknowledge is received, next file part can be sent */
198 if ( u16FileAccessState == FOE_WAIT_FOR_ACK )
199 {
200 /* we are waiting for an acknowledge, service is correct, call the application function
201 to get the next part of the file */
202/* ECATCHANGE_START(V5.12)*/
203 nextState = FOE_Ack( u32FileOffset, (UINT16 MBXMEM *)pFoeInd->Data );
204/* ECATCHANGE_END(V5.12)*/
205 /* u32LastFileOffset contains the offset of the file which is sent now */
206 u32LastFileOffset = u32FileOffset;
207
208 if (nextState <= FOE_MAXDATA)
209 {
210 /* u32FileOffset contains the offset of the file which shall be sent when the next FoE ACK is received */
211 u32FileOffset += nextState;
212 /* increment the packet number */
213 u32PacketNo++;
214 }
215 }
216 else if ( u16FileAccessState == FOE_WAIT_FOR_LAST_ACK )
217 {
218 /* we were waiting for the last acknowledge, now the sequence is finished */
219 nextState = FOE_FINISHED_NOACK;
220 }
221 break;
222
223 case ECAT_FOE_OPCODE_ERR:
224 /* a FoE Error service is received */
225 if ( u16FileAccessState != FOE_READY )
226 {
227/*ECATCHANGE_START(V5.12) FOE2*/
228 UINT32 u32CmdErrorCode = SWAPDWORD((((UINT32)pFoeInd->FoeHeader.Cmd.ErrorCode[FOE_COMMAND_LOWWORD]) | ((UINT32)pFoeInd->FoeHeader.Cmd.ErrorCode[FOE_COMMAND_HIGHWORD]) << 16));
229/*ECATCHANGE_END(V5.12) FOE2*/
230
231 /* a file transmission sequence is active, inform the application, that this sequence
232 was stopped */
233 FOE_Error( u32CmdErrorCode );
234
235 nextState = FOE_FINISHED;
236 }
237 break;
238
239 case ECAT_FOE_OPCODE_BUSY:
240 /* a FoE Busy service is received */
241 if ( u16FileAccessState == FOE_WAIT_FOR_ACK
242 || u16FileAccessState == FOE_WAIT_FOR_LAST_ACK )
243 {
244 /* we are waiting for an acknowledge, service is correct, call the application function
245 to resend the last part of the file */
246/* ECATCHANGE_START(V5.12)*/
247 nextState = FOE_Busy( SWAPWORD(pFoeInd->FoeHeader.Cmd.Busy.Done), u32LastFileOffset, (UINT16 MBXMEM *)pFoeInd->Data );
248/* ECATCHANGE_END(V5.12)*/
249 }
250 break;
251
252 }
253
254 if ( nextState <= FOE_MAXDATA )
255 {
256 /* we send DATA and wait for ACK */
257 UINT32 d = SWAPDWORD(u32PacketNo);
258
259 /* store the OpCode in the mailbox buffer */
260 pFoeInd->FoeHeader.OpCode = SWAPWORD(ECAT_FOE_OPCODE_DATA);
261 /* store the packet number in the mailbox buffer */
262/*ECATCHANGE_START(V5.12) FOE2*/
263 pFoeInd->FoeHeader.Cmd.PacketNo[FOE_COMMAND_LOWWORD] = ((d & 0x0000FFFF));
264 pFoeInd->FoeHeader.Cmd.PacketNo[FOE_COMMAND_HIGHWORD] = ((d & 0xFFFF0000) >> 16);
265/*ECATCHANGE_END(V5.12) FOE2*/
266
267 /* store the size of the mailbox data in the mailbox buffer */
268 pFoeInd->MbxHeader.Length = FOE_HEADER_SIZE + nextState;
269
270 if ( nextState == u16SendMbxSize - FOE_HEADER_SIZE - MBX_HEADER_SIZE )
271 {
272 /* packets still following, we wait for an ACK */
273 u16FileAccessState = FOE_WAIT_FOR_ACK;
274 }
275 else
276 {
277 /* it was the last Packet, we wait for the last ACK */
278 u16FileAccessState = FOE_WAIT_FOR_LAST_ACK;
279 }
280 }
281 else if ( nextState <= FOE_MAXBUSY )
282 {
283 /* we are still storing the received file data (in flash for example) and
284 send BUSY and wait for the DATA to be sent again */
285 /* store the OpCode in the mailbox buffer */
286 pFoeInd->FoeHeader.OpCode = SWAPWORD(ECAT_FOE_OPCODE_BUSY);
287 /* store the information how much progress we made until we can receive file data again */
288 pFoeInd->FoeHeader.Cmd.Busy.Done = SWAPWORD(nextState-FOE_MAXBUSY_ZERO);
289
290 pFoeInd->FoeHeader.Cmd.Busy.Entire = 0;
291 /* store the size of the mailbox data in the mailbox buffer */
292
293 pFoeInd->MbxHeader.Length = FOE_HEADER_SIZE;
294 }
295 else if ( nextState == FOE_ACK || nextState == FOE_ACKFINISHED )
296 {
297 /* we send ACK and wait for DATA
298 the next file data is expected with an incremented packet number, but
299 we have to acknowledge the old packet first */
300 UINT32 d = SWAPDWORD(u32PacketNo);
301 u32PacketNo++;
302
303 /* store the OpCode in the mailbox buffer */
304 pFoeInd->FoeHeader.OpCode = SWAPWORD(ECAT_FOE_OPCODE_ACK);
305 /* store the packet number in the mailbox buffer */
306/*ECATCHANGE_START(V5.12) FOE2*/
307 pFoeInd->FoeHeader.Cmd.PacketNo[FOE_COMMAND_LOWWORD] = ((d & 0x0000FFFF));
308 pFoeInd->FoeHeader.Cmd.PacketNo[FOE_COMMAND_HIGHWORD] = ((d & 0xFFFF0000) >> 16);
309/*ECATCHANGE_END(V5.12) FOE2*/
310
311 /* store the size of the mailbox data in the mailbox buffer */
312 pFoeInd->MbxHeader.Length = SIZEOF(TFOEHEADER);
313
314 /* we wait for the next data part */
315 if ( nextState == FOE_ACK )
316 {
317 /* we wait for the next data part */
318 u16FileAccessState = FOE_WAIT_FOR_DATA;
319 }
320 else
321 {
322 /* the last data part was received */
323 u16FileAccessState = FOE_READY;
324 }
325 }
326 else if ( nextState < FOE_ERROR )
327 {
328 /* the file transmission sequence is finished, we have to send nothing */
329 u16FileAccessState = FOE_READY;
330 APPL_FreeMailboxBuffer(pFoeInd);
331 return 0;
332 }
333 else
334 {
335 UINT32 d = SWAPDWORD(nextState);
336 UINT8 b = 0;
337
338 /* store the OpCode in the mailbox buffer */
339 pFoeInd->FoeHeader.OpCode = SWAPWORD(ECAT_FOE_OPCODE_ERR);
340 /* store the ErrorCode in the mailbox buffer */
341/*ECATCHANGE_START(V5.12) FOE2*/
342 pFoeInd->FoeHeader.Cmd.ErrorCode[FOE_COMMAND_LOWWORD] = ((d & 0x0000FFFF));
343 pFoeInd->FoeHeader.Cmd.ErrorCode[FOE_COMMAND_HIGHWORD] = ((d & 0xFFFF0000) >> 16);
344/*ECATCHANGE_END(V5.12) FOE2*/
345
346 /* store the size of the mailbox data in the mailbox buffer */
347 pFoeInd->MbxHeader.Length = SIZEOF(TFOEHEADER);
348
349 /* Error Text is returned in pFoeInd->Data */
350 while (b < 32)
351 {
352 UINT16 data = ((UINT16 MBXMEM *) pFoeInd->Data)[(b >> 1)];
353
354 if ((data & 0x00FF) == 0)
355 {
356 break;
357 }
358 b++;
359
360 if ((data & 0xFF00) == 0)
361 {
362 break;
363 }
364 b++;
365 }
366 if ( b < 32 )
367 {
368 pFoeInd->MbxHeader.Length += b+1;
369 }
370
371 /* the file transmission sequence is finished */
372 u16FileAccessState = FOE_READY;
373 }
374
375 if ( bBootMode )
376 {
377 /* in BOOT mode the mailbox buffer is not sent via the mailbox functions
378 because only FoE is allowed in BOOT mode, so we have to include the
379 mailbox data link layer counter */
380 pFoeInd->MbxHeader.Flags[MBX_OFFS_COUNTER] &= ~MBX_MASK_COUNTER;
381 if ( (u8MbxReadCounter & 0x07) == 0 )
382 {
383 /* counter 0 is not allowed if mailbox data link layer is supported */
384 u8MbxReadCounter = 1;
385 }
386 /* store the counter in the mailbox header */
387 pFoeInd->MbxHeader.Flags[MBX_OFFS_COUNTER] |= u8MbxReadCounter << MBX_SHIFT_COUNTER;
388 /* increment the counter for the next service */
389 u8MbxReadCounter++;
390 /* call the function to send the mailbox service directly,
391 in BOOT mode we can be sure that the send mailbox is empty
392 because no parallel services are allowed */
393 MBX_CopyToSendMailbox((TMBX MBXMEM *) pFoeInd);
394 }
395 else
396 {
397 if ( MBX_MailboxSendReq((TMBX MBXMEM *) pFoeInd, FOE_SERVICE) != 0 )
398 {
399 /* if the mailbox service could not be sent (or stored), the response will be
400 stored in the variable pFoeSendStored and will be sent automatically
401 from the mailbox handler (FOE_ContinueInd) when the send mailbox will be read
402 the next time from the master */
403 pFoeSendStored = (TMBX MBXMEM *) pFoeInd;
404 }
405 }
406
407 return 0;
408}
