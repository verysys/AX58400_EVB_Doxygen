51/**
52 * @brief Configure the MPU attributes as Not cacheable for D3 SRAM.
53 * @note The Base Address is D3_SRAM_BASE.
54 * The Region Size is 64KB.
55 * @param None
56 * @retval None
57 */
58static void MPU_Config(void)
59{
60 MPU_Region_InitTypeDef MPU_InitStruct;
61
62 /* Disable the MPU */
63 HAL_MPU_Disable();
64
65 /* Configure the MPU attributes as WT for SDRAM */
66 MPU_InitStruct.Enable = MPU_REGION_ENABLE;
67 MPU_InitStruct.BaseAddress = D3_SRAM_BASE;
68 MPU_InitStruct.Size = MPU_REGION_SIZE_64KB;
69 MPU_InitStruct.AccessPermission = MPU_REGION_FULL_ACCESS;
70 MPU_InitStruct.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
71 MPU_InitStruct.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
72 MPU_InitStruct.IsShareable = MPU_ACCESS_SHAREABLE;
73 MPU_InitStruct.Number = MPU_REGION_NUMBER0;
74 MPU_InitStruct.TypeExtField = MPU_TEX_LEVEL0;
75 MPU_InitStruct.SubRegionDisable = 0x00;
76 MPU_InitStruct.DisableExec = MPU_INSTRUCTION_ACCESS_ENABLE;
77
78 HAL_MPU_ConfigRegion(&MPU_InitStruct);
79
80 /* Enable the MPU */
81 HAL_MPU_Enable(MPU_PRIVILEGED_DEFAULT);
82}
