92 \param pName Pointer to the name of the file
93 \param nameSize Length of the file name
94 \param pData Pointer to the response data
95 \param password Password for the file read
96
97 \return size of the data to be sent, busy or an error code
98 1..FOE_MAXBUSY-101 (size of data to be sent)
99 FOE_MAXBUSY-100 (0%) (busy)
100 FOE_MAXBUSY (100%) (busy)
101 ECAT_FOE_ERRCODE_NOTDEFINED (error)
102 ECAT_FOE_ERRCODE_NOTFOUND (error)
103 ECAT_FOE_ERRCODE_ACCESS (error)

110 \brief The function starts a file reading
111
112*////////////////////////////////////////////////////////////////////////////////////////
113UINT16 AX58400_FoeRead(UINT16 MBXMEM * pName, UINT16 nameSize, UINT32 password, UINT16 maxBlockSize, UINT16 *pData)
114{
115 UINT16 size = 0;
116 UINT16 i = 0;
117 CHAR aReadFileName[MAX_FILE_NAME_SIZE];
118
119 /* ECATCHANGE_START(V5.11) FOE2*/
120 if ((nameSize + 1) > MAX_FILE_NAME_SIZE)
121 {
122 return ECAT_FOE_ERRCODE_DISKFULL;
123 }
124
125 /*Read requested file name to endianess conversion if required*/
126 MBXSTRCPY(aReadFileName, pName, nameSize);
127 aReadFileName[nameSize] = '\0';
128
129 /* ECATCHANGE_END(V5.11) FOE2*/
130 {
131 /* for test only the written file name can be read */
132 for (i = 0; i < nameSize; i++)
133 {
134 if (aReadFileName[i] != aFileName[i])
135 {
136 /* file name not found */
137 return ECAT_FOE_ERRCODE_NOTFOUND;
138 }
139 }
140
141 if (nFileSize >= (u16SendMbxSize - SIZEOF(TFOEHEADER) - MBX_HEADER_SIZE))
142 {
143 size = (u16SendMbxSize - SIZEOF(TFOEHEADER) - MBX_HEADER_SIZE);
144 }
145 else
146 {
147 size = (unsigned short) nFileSize;
148 }
149 MBXMEMCPY(pData, aFileData, size);
150 }
151 return size;
152}

172 If the slave stack requires some time to handle the incoming data the function FOE_Data() should return busy)

174 \brief The function starts a file writing
175
176*////////////////////////////////////////////////////////////////////////////////////////
177UINT16 AX58400_FoeWrite(UINT16 MBXMEM * pName, UINT16 nameSize, UINT32 password)
178{
179 if ((nameSize >= SIZEOF(aFirmwareDownloadHeader))
180 &&(pName[0] == aFirmwareDownloadHeader[0])
181 &&(pName[1] == aFirmwareDownloadHeader[1])
182 &&(pName[2] == aFirmwareDownloadHeader[2])
183 &&(pName[3] == aFirmwareDownloadHeader[3]))
184 {
185 if (bBootMode)
186 {
187 BL_StartDownload(password);
188 return 0;
189 }
190 else
191 {
192 return ECAT_FOE_ERRCODE_BOOTSTRAPONLY;
193 }
194 }
195 else if (bBootMode)
196 {
197 return ECAT_FOE_ERRCODE_NOTINBOOTSTRAP;
198 }
199 else if (nameSize < MAX_FILE_NAME_SIZE)
200 {
201 /* for test every file name can be written */
202 MBXSTRCPY(aFileName, pName, nameSize);
203 MBXSTRCPY(aFileName+nameSize, "\0", 1); //string termination
204
205 nFileWriteOffset = 0;
206 nFileSize = 0;
207 return 0;
208 }
209
210 return ECAT_FOE_ERRCODE_DISKFULL;
211}

231 \brief The function is called to write the next part of a file
232
233*////////////////////////////////////////////////////////////////////////////////////////
234UINT16 AX58400_FoeWriteData(UINT16 MBXMEM * pData, UINT16 Size, BOOL bDataFollowing)
235{
236 if (bBootMode)
237 {
238 return BL_Data(pData, Size);
239 }
240 else if ((nFileWriteOffset + Size) > MAX_FILE_SIZE)
241 {
242 return ECAT_FOE_ERRCODE_DISKFULL;
243 }
244
245 if (Size)
246 {
247 MBXMEMCPY(&aFileData[(nFileWriteOffset >> 1)], pData, Size);
248 }
249
250 if (Size == (u16ReceiveMbxSize - MBX_HEADER_SIZE - FOE_HEADER_SIZE))
251 {
252 /* FoE-Data services will follow */
253 nFileWriteOffset += Size;
254 return 0;
255 }
256 else
257 {
258 /* last part of the file is written */
259 nFileSize = nFileWriteOffset + Size;
260 nFileWriteOffset = 0;
261 return FOE_ACKFINISHED;
262 }
263}

284 \brief The function is called when the reading of a part of a file is acknowledged
285
286*////////////////////////////////////////////////////////////////////////////////////////
287
288UINT16 AX58400_FoeReadData(UINT32 offset, UINT16 maxBlockSize, UINT16 *pData)
289{
290 if (offset < nFileSize)
291 {
292 /* send next part of the file */
293 UINT32 size;
294 UINT32 sendSize = nFileSize - offset;
295
296 if (sendSize >= (u16SendMbxSize - SIZEOF(TFOEHEADER) - MBX_HEADER_SIZE))
297 {
298 size = (u16SendMbxSize - SIZEOF(TFOEHEADER) - MBX_HEADER_SIZE);
299 }
300 else
301 {
302 size = sendSize;
303 }
304
305 MBXMEMCPY(pData, &aFileData[(offset >> 1)], size);
306
307 return ((UINT16) size);
308 }
309 else if (offset == nFileSize)
310 {
311 /* file transfer is finished */
312 return 0; // size = 0
313 }
314 return ECAT_FOE_ERRCODE_ILLEGAL;
315}
