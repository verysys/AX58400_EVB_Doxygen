82/*
83 * ----------------------------------------------------------------------------
84 * Function Name: Cmd_CM4_RecvCmd()
85 * Purpose:
86 * Params:
87 * Returns:
88 * Note:
89 * ----------------------------------------------------------------------------
90 */
91int32_t Cmd_CM4_RecvCmd(void)
92{
93 static int32_t AckCnt = 0;
94 pTCiA402Axis pCiA402Axis;
95 /* Check message id */
96 if ((pAmpMsgReq->MsgId != pAmpMsgRsp->MsgId))
97 {
98 pCmdTableRsp = (CMD_TABLE *)pAmpMsgRsp->Data;
99 memset((void*)pCmdTableRsp, 0, sizeof(CMD_TABLE));
100 switch (pAmpMsgReq->MsgType)
101 {
102 case AMPCMD_CMD_RUN:
103 pCmdTable = (CMD_TABLE*)&(pAmpMsgReq->Data);
104 //AMPCMD_memcpy((uint8_t *)pCmdTable,(uint8_t *)&(pAmpMsgReq->Data), pAmpMsgReq->ByteLen);
105 AMPCMD_memcpy((uint8_t *)&pCiA402Axis,(uint8_t *)&(pCmdTable->CiA402AxisAddr), sizeof(pCmdTable->CiA402AxisAddr));
106 if(pCmdTable->GetFaultAckFlag)
107 {
108 pCiA402Axis->u8FaultAck = MCI_FaultAcknowledged(pCiA402Axis->pMCI);
109 if(AckCnt%20 == 0)
110 {
111 AckCnt = 1;
112 }
113 else
114 AckCnt++;
115 if (pCiA402Axis->u8FaultAck == 1)
116 {
117 pCiA402Axis->Objects.objErrorCode = 0;
118 pCiA402Axis->i16State = STATE_SWITCH_ON_DISABLED;// Transition 15
119 }
120 pCmdTableRsp->GetFaultAckFlag = 1;
121 }
122 if(pCmdTable->InitCIA402Flag)
123 {
124 pCiA402Axis->pMCI = (void *)pMCI[pCiA402Axis->AxisCnt];
125 AMPCMD_memcpy((uint8_t *)pCiA402Axis->pMCI,(uint8_t *)pMCI[pCiA402Axis->AxisCnt], sizeof(void *));
126 pCmdTableRsp->InitCIA402Flag = 1;
127 }
128 if(pCmdTable->GetMecAvgSpdFlag)
129 {
130 pCmdTableRsp->GetMecAvgSpdFlag = 1;
131 if(pCiA402Axis->pMCI)
132 {
133 pCiA402Axis->Objects.objVelocityActualValue = MCI_GetAvrgMecSpeedUnit(pCiA402Axis->pMCI)*6;
134 }
135 }
136 if(pCmdTable->GetIQDFlag)
137 {
138 INT16 TorqueActualValue = MCI_GetIqd(pCiA402Axis->pMCI).q;
139 pCiA402Axis->Objects.objTorqueActualValue = TorqueActualValue;
140 pCmdTableRsp->GetIQDFlag = 1;
141 }
142 if(pCmdTable->GetHallStuFlag)
143 {
144 pCiA402Axis->u16HallStatus = CIA402_GetHallStatus();
145 pCmdTableRsp->GetHallStuFlag = 1;
146 }
147 if(pCmdTable->GetRstPosFlag)
148 {
149 //POS_ResetPosition
150 PID_HandleInit(&PID_PosParamsM1);
151 pPosCtrlM1.EncoderAbsoluteAligned = false;
152 TC_EncoderReset(&pPosCtrlM1);
153 pCmdTableRsp->GetRstPosFlag = 1;
154 }
155 if(pCmdTable->GetActPosFlag)
156 {
157 pCiA402Axis->Objects.objPositionActualValue = (INT32)(((double)(MC_GetCurrentPosition1())/MOTOR_RATIO)+(double)0.5);//POS_GetActualPosition();
158 pCmdTableRsp->GetActPosFlag = 1;
159 }
160 if(pCmdTable->StpMCFlag)
161 {
162 MCI_StopMotor(pCiA402Axis->pMCI);
163 pCmdTableRsp->StpMCFlag = 1;
164 }
165 if(pCmdTable->StaMCFlag)
166 {
167 MCI_StartMotor(pCiA402Axis->pMCI);
168 pCmdTableRsp->StaMCFlag = 1;
169 }
170
171 if(pCmdTable->GetCurFaultFlag)
172 {
173 pCiA402Axis->u16CurFault = MCI_GetOccurredFaults(pCiA402Axis->pMCI);
174 pCmdTableRsp->GetCurFaultFlag = 1;
175 }
176
177 if(pCmdTable->GetMCSteFlag)
178 {
179 pCiA402Axis->i32STMState = MCI_GetSTMState(pCiA402Axis->pMCI);
180 pCmdTableRsp->GetMCSteFlag = 1;
181 }
182
183 if(pCmdTable->SetCIA402FuncFlag)
184 {
185 switch (pCiA402Axis->Objects.objModesOfOperationDisplay)
186 {
187 case HOMING_MODE:
188 pCiA402Axis->pFuncIntf = &CIA402HM_FuncIntf;
189
190 break;
191 case CYCLIC_SYNC_POSITION_MODE:
192 pCiA402Axis->pFuncIntf = &CIA402CSP_FuncIntf;
193 break;
194 case PROFILE_POSITION_MODE:
195 pCiA402Axis->pFuncIntf = &CIA402PP_FuncIntf;;
196 break;
197 default:
198 pCiA402Axis->pFuncIntf = 0;
199 break;
200 }
201 pCmdTableRsp->SetCIA402FuncFlag = 1;
202 }
203 if(pCmdTable->StaCIA402FuncFlag)
204 {
205 if(pCiA402Axis->pFuncIntf)
206 {
207 pCiA402Axis->pFuncIntf->FuncStart(pCiA402Axis);
208 }
209 pCmdTableRsp->StaCIA402FuncFlag = 1;
210 }
211 if(pCmdTable->RunCIA402FuncFlag)
212 {
213 if(pCiA402Axis->pFuncIntf)
214 {
215 pCiA402Axis->pFuncIntf->FuncRun(pCiA402Axis);
216 }
217 pCmdTableRsp->RunCIA402FuncFlag = 1;
218 }
219 if(pCmdTable->StpCIA402FuncFlag)
220 {
221 if(pCiA402Axis->pFuncIntf)
222 {
223 pCiA402Axis->pFuncIntf->FuncStop(pCiA402Axis);
224
225 }
226 pCmdTableRsp->StpCIA402FuncFlag = 1;
227 }
228
229 pAmpMsgRsp->ByteLen = sizeof(CMD_TABLE);
230 break;
231 default:
232
233 break;
234 }
235 pAmpMsgRsp->MsgId = pAmpMsgReq->MsgId;
236 pAmpMsgRsp->MsgType = pAmpMsgReq->MsgType | AMPCMD_ACK_MASK;
237 }
238
239
240 /* Ack to Master CPU */
241 HAL_HSEM_FastTake(HSEM_ID_1);// Take HSEM
242 HAL_HSEM_Release(HSEM_ID_1,0);//Release HSEM in order to notify the CPU1(CM7)
243
244 return -1;
245}
