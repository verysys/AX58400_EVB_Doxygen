89/////////////////////////////////////////////////////////////////////////////////////////
90/**
91 \param pCoeMbx Pointer to the received mailbox data from the master.
92
93 \return result of the operation (0 (success) or mailbox error code (MBXERR_.... defined in
94 mailbox.h))
95
96 \brief This function is called when a CoE (CAN application layer over EtherCAT) service is received from
97 the master.
98*////////////////////////////////////////////////////////////////////////////////////////
99
100UINT8 COE_ServiceInd(TCOEMBX MBXMEM *pCoeMbx)
101{
102 UINT8 result = 0;
103
104 switch ((pCoeMbx->CoeHeader & COEHEADER_COESERVICEMASK) >> COEHEADER_COESERVICESHIFT)
105 {
106 case COESERVICE_SDOREQUEST:
107 /* SDO-Request received, call SDOS_SdoInd to process the SDO-Request
108 if an existing SDO-Stack shall be used, the corresponding function
109 should be called */
110 result = SDOS_SdoInd( (TINITSDOMBX MBXMEM *) pCoeMbx );
111 break;
112
113 case COESERVICE_SDOINFO:
114 /* SDO-Information Request received, call SDOS_SdoInfoInd to process the SDO-Request */
115 result = SDOS_SdoInfoInd( (TSDOINFORMATION MBXMEM *) pCoeMbx );
116 /*NOERROR_INWORK is never returned by SDOS_SdoInfoInd() => delete return code handling*/
117 break;
118
119
120 case COESERVICE_SDORESPONSE:
121 case COESERVICE_EMERGENCY:
122 case COESERVICE_TXPDO:
123 case COESERVICE_RXPDO:
124 case COESERVICE_TXPDOREMREQ:
125 case COESERVICE_RXPDOREMREQ:
126 /* these CoE services are not supported yet */
127 result = MBXERR_SERVICENOTSUPPORTED;
128 break;
129
130 default:
131 result = MBXERR_INVALIDHEADER;
132 break;
133 }
134 return result;
135}
136
137/////////////////////////////////////////////////////////////////////////////////////////
138/**
139 \param pMbx Pointer to the free mailbox to sent.
140
141 \return result of the operation (0 (success)
142
143 \brief This function is called when a CoE service to be sent is stored and can
144 \brief be put in the send mailbox.
145*////////////////////////////////////////////////////////////////////////////////////////
146
147UINT8 COE_ContinueInd(TMBX MBXMEM * pMbx)
148{
149 if (pCoeSendStored)
150 {
151 /* send the stored CoE service which could not be sent before */
152 MBX_MailboxSendReq(pCoeSendStored, 0);
153 pCoeSendStored = 0;
154 }
155 else
156 {
157 /* send the next fragment of the last CoE service (only for SDO-Information possible) */
158 /* in mailbox queue mode pMbx is always 0, so a mailbox buffer shall be get */
159 pMbx = (TMBX MBXMEM *) APPL_AllocMailboxBuffer(SIZEOF(TMBX));
160 /* it shall be checked if a valid pointer was returned */
161 if (pMbx == NULL)
162 {
163 return MBXERR_NOMOREMEMORY;
164 }
165 else
166 {
167 /* copy the stored SDO-Info-Header in the request */
168 MBXMEMCPY(pMbx, aSdoInfoHeader, SDO_INFO_HEADER_BYTE_SIZE);
169 /* call SDOS_SdoInfoInd to generate and send the next fragment */
170 SDOS_SdoInfoInd( (TSDOINFORMATION MBXMEM *) pMbx );
171 }
172 }
173
174 return 0;
175}
